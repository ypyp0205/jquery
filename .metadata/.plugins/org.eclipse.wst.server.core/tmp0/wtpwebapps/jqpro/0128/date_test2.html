살아온 날 구하기

		today = new Date();
		myday = new Date(1990, 02, 05);

		time = today.getTime() - myday.getTime();

		// alert(time);
		sec = time / 1000;
		min = sec / 60;
		hour = min / 60;
		day = hour / 24;
		year = day / 365;
